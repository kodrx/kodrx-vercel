Emitir Receta Médica

const db = firebase.database(); function guardarReceta() { const medico = document.getElementById('medico').value; const medicamento = document.getElementById('medicamento').value; const dosis = document.getElementById('dosis').value; const dias = document.getElementById('dias').value; const instrucciones = document.getElementById('instrucciones').value; if (!medico || !medicamento || !dosis || !dias) { alert("Por favor completa todos los campos obligatorios."); return; } const receta = { medico, medicamento, dosis, duracion: dias, instrucciones, fecha: new Date().toISOString() }; const nuevaRef = db.ref('recetas').push(); nuevaRef.set(receta).then(() => { const recetaID = nuevaRef.key; const url = `https://kodrx-vercel.vercel.app/verificar.html?id=${recetaID}`; const qr = `https://api.qrserver.com/v1/create-qr-code/?data=${url}&size=200x200`; document.getElementById('resultado').innerHTML = `

QR generado:


${url} `; }).catch((error) => { alert('Error al guardar la receta: ' + error.message); }); }
